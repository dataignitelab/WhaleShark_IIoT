[docs]def str2hex(arg_str):
    #String to hex
    fun_id = arg_str.encode('ASCII')

    hex_fun_id = hex_fun_id.decode('ASCII')

    #Hex to String
    fun_id_byte = bytes.fromhex(arg_hex)
    fun_id_str = fun_id_byte.decode('ASCII')
    return fun_id_str
